[docs]def distribution_radius_check(center, radius, N):
    angles = (np.random.rand(N) - 0.5) * 2 * radius
    angles += center
    S = np.vstack((np.cos(angles), np.sin(angles)))
    r2 = distribution_radius(S)
    check_reasonable_radius(radius, r2, N)
