def SO2_project_from_SO3(b):
    direction = np.array([1, 0, 0])
    vector = np.dot(b, direction)
    n = np.linalg.norm(vector)
    atol = 1e-8  # XXX: make common
    if n < atol:
        return rot2d(0)
    else:
        theta = np.arctan2(vector[1], vector[0])
        return rot2d(theta)
