[docs]@contract(A='array[NxK]')
def normalize_columns(A):
    A = A.copy()
    _, k = A.shape
    for j in range(k):
        A[:, j] = A[:, j] / np.linalg.norm(A[:, j])
    return A




[docs]@contract(A='array[NxK]')
def proj_from_subspace(A):
    An = normalize_columns(A)
    return np.dot(An, An.T)
