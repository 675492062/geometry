@contract(X='array[KxN],K>=2,K<N', Y='array[KxN]', returns='array[KxK],orthogonal')

[docs]def best_orthogonal_transform(X, Y):
    ''' Finds the best orthogonal transform R  between X and Y,
        such that R X ~= Y. '''
    YX = dot(Y, X.T)
    check('array[KxK]', YX)
    U, S, V = svd(YX) #@UnusedVariable
    best = dot(U, V)
    return best
